Первое задание:
SELECT c."login" AS CourierLogin,
	    COUNT(o."id") AS OrdersCount
FROM "Couriers" AS c
INNER JOIN "Orders" AS o ON c."id" = o."courierId"
WHERE o."inDelivery" = true
GROUP BY c."login";
